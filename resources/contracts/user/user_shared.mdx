<Expandable title="Implementation">
  ```rust
    public fun has_analytics(
        shared_user: &mut UserShared,
        app_address: address,
        epoch: u64,
    ): bool {
        let analytics_key = AnalyticsKey {
            app: app_address,
            epoch
        };

        dof::exists_with_type<AnalyticsKey, Analytics>(
            &shared_user.id,
            analytics_key
        )
    }
  ```
</Expandable>
